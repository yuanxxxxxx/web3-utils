# Web3 集成指南

## 快速开始

### 1. 配置链和钱包

```typescript
// lib/web3/config.ts
import { sepolia } from 'wagmi/chains'

export const config = getDefaultConfig({
  appName: 'Your App',
  projectId: 'YOUR_WALLETCONNECT_PROJECT_ID',
  chains: [sepolia],
  connectors: [metaMaskWallet, rainbowWallet, okxWallet]
})
```

### 2. 在 Layout 中包裹 Provider

```tsx
// app/layout.tsx
import Web3Provider from '@/lib/web3/Web3Provider'

export default function RootLayout({ children }) {
  return (
    <Web3Provider>
      {children}
    </Web3Provider>
  )
}
```

## 核心 Hooks

### useActiveWeb3React - 获取连接状态

```tsx
import useActiveWeb3React from '@/hooks/useActiveWeb3React'

const { 
  account,           // 当前地址
  chainId,          // 当前链 ID
  client,           // Viem Client
  isConnecting,     // 连接状态
  openConnectModal, // 打开连接弹窗
  loginOut          // 登出（断开钱包+清除状态）
} = useActiveWeb3React()

// 示例：显示地址
<p>{account || 'Not Connected'}</p>

// 示例：连接钱包
<button onClick={openConnectModal}>Connect Wallet</button>

// 示例：登出
<button onClick={loginOut}>Logout</button>
```

## 读取合约数据

### useReadContracts - 批量读取

```tsx
import { useReadContracts } from 'wagmi'
import { erc20Abi } from 'viem'

const { data, refetch } = useReadContracts({
  contracts: [
    {
      address: '0xTokenAddress',
      abi: erc20Abi,
      functionName: 'balanceOf',
      args: [account as `0x${string}`]
    },
    {
      address: '0xTokenAddress',
      abi: erc20Abi,
      functionName: 'allowance',
      args: [account, spender]
    }
  ],
  query: {
    enabled: !!account // 仅在有地址时查询
  }
})

// 获取结果
const balance = data?.[0]?.result as bigint
const allowance = data?.[1]?.result as bigint

// 手动刷新
<button onClick={() => refetch()}>Refresh</button>
```

## 发送交易

### useWriteContract + waitForTransactionReceipt

```tsx
import { useWriteContract } from 'wagmi'
import { waitForTransactionReceipt } from 'viem/actions'
import { erc20Abi } from 'viem'
import useActiveWeb3React from '@/hooks/useActiveWeb3React'
import toast from 'react-hot-toast'

const { client } = useActiveWeb3React()
const { mutateAsync: writeContract } = useWriteContract()

const handleApprove = async () => {
  try {
    // 1. 发送交易
    const hash = await writeContract({
      address: '0xTokenAddress',
      abi: erc20Abi,
      functionName: 'approve',
      args: ['0xSpender', BigInt(1000000000000000000)]
    })

    toast.loading('Transaction pending...', { id: 'tx' })

    // 2. 等待确认
    const receipt = await waitForTransactionReceipt(client, {
      hash,
      confirmations: 1,
      pollingInterval: 1000
    })

    // 3. 检查状态
    if (receipt.status === 'success') {
      toast.success('Approved!', { id: 'tx' })
    } else {
      toast.error('Failed', { id: 'tx' })
    }
  } catch (error) {
    console.error(error)
    toast.error('Transaction failed', { id: 'tx' })
  }
}
```

## 完整示例

```tsx
'use client'

import useActiveWeb3React from '@/hooks/useActiveWeb3React'
import { useReadContracts, useWriteContract } from 'wagmi'
import { erc20Abi } from 'viem'
import { waitForTransactionReceipt } from 'viem/actions'
import toast from 'react-hot-toast'

export default function TokenPage() {
  const { account, client } = useActiveWeb3React()
  const { mutateAsync: writeContract } = useWriteContract()
  
  const tokenAddress = '0x...'
  
  // 读取余额
  const { data } = useReadContracts({
    contracts: [{
      address: tokenAddress,
      abi: erc20Abi,
      functionName: 'balanceOf',
      args: [account as `0x${string}`]
    }],
    query: { enabled: !!account }
  })
  
  const balance = data?.[0]?.result as bigint
  
  // 发送 Approve
  const handleApprove = async () => {
    try {
      const hash = await writeContract({
        address: tokenAddress,
        abi: erc20Abi,
        functionName: 'approve',
        args: ['0xSpender', BigInt(1e18)]
      })
      
      await waitForTransactionReceipt(client, { hash })
      toast.success('Approved!')
    } catch (error) {
      toast.error('Failed')
    }
  }
  
  return (
    <div>
      <p>Balance: {balance?.toString()}</p>
      <button onClick={handleApprove}>Approve</button>
    </div>
  )
}
```
