# 组件使用指南

## 弹窗 - CModal

### 基础用法

```tsx
import CModal from '@/components/Basic/CModal'
import { useState } from 'react'

const [visible, setVisible] = useState(false)

<CModal 
  visible={visible}
  onClose={() => setVisible(false)}
  title="Modal Title"
>
  <div>Modal Content</div>
</CModal>
```

### Props

| 参数 | 说明 | 类型 | 默认值 |
|-----|------|-----|--------|
| `visible` | 是否显示 | `boolean` | - |
| `onClose` | 关闭回调 | `() => void` | - |
| `title` | 标题 | `React.ReactNode` | - |
| `width` | 宽度 | `string \| number` | `'446px'` |
| `showHeader` | 显示头部 | `boolean` | `true` |
| `padding` | 内容区 padding | `string` | `'0'` |
| `zIndex` | 层级 | `number` | `99` |

## 分页器 - Pagination

### 基础用法

```tsx
import Pagination from '@/components/Basic/Pagination'
import { useState } from 'react'

const [current, setCurrent] = useState(1)

<Pagination
  current={current}
  total={100}
  pageSize={10}
  onChange={(page) => setCurrent(page)}
/>
```

### Props

| 参数 | 说明 | 类型 | 默认值 |
|-----|------|-----|--------|
| `current` | 当前页 | `number` | - |
| `total` | 总条数 | `number` | - |
| `pageSize` | 每页条数 | `number` | `10` |
| `onChange` | 页码改变回调 | `(page: number) => void` | - |
| `showTotal` | 显示总数 | `boolean` | `true` |
| `size` | 尺寸 | `'default' \| 'small'` | `'default'` |

## 下拉菜单 - Dropdown

### 基础用法

```tsx
import Dropdown from '@/components/Basic/Dropdown'
import { useState } from 'react'

const [value, setValue] = useState('1')

<Dropdown
  value={value}
  options={[
    { label: 'Option 1', value: '1' },
    { label: 'Option 2', value: '2' }
  ]}
  onChange={(val) => setValue(val)}
/>
```

### Props

| 参数 | 说明 | 类型 | 默认值 |
|-----|------|-----|--------|
| `value` | 当前值 | `string` | - |
| `options` | 选项列表 | `{label: string, value: string}[]` | - |
| `onChange` | 值改变回调 | `(value: string) => void` | - |
| `placeholder` | 占位文本 | `string` | `'Select'` |
| `fullWidth` | 全宽显示 | `boolean` | `false` |
| `minimal` | 极简模式 | `boolean` | `false` |

## Toast 通知

### 基础用法

```tsx
import toast from 'react-hot-toast'

// 成功
toast.success('Success message')

// 错误
toast.error('Error message')

// 加载中
toast.loading('Loading...', { id: 'loading' })

// 更新 toast
toast.success('Done!', { id: 'loading' })
```

### 常见场景

```tsx
// 交易提示
const handleTransaction = async () => {
  try {
    toast.loading('Transaction pending...', { id: 'tx' })
    const hash = await sendTransaction()
    await waitForReceipt(hash)
    toast.success('Transaction confirmed!', { id: 'tx' })
  } catch (error) {
    toast.error('Transaction failed', { id: 'tx' })
  }
}
```

## 完整示例页面

参考 `app/example/page.tsx`:

```tsx
'use client'

import { useState } from 'react'
import CModal from '@/components/Basic/CModal'
import CTooltip from '@/components/Basic/CTooltip'
import Pagination from '@/components/Basic/Pagination'
import Dropdown from '@/components/Basic/Dropdown'

export default function ComponentsDemo() {
  const [showModal, setShowModal] = useState(false)
  const [page, setPage] = useState(1)
  const [value, setValue] = useState('1')

  return (
    <div>
      <button onClick={() => setShowModal(true)}>
        Open Modal
      </button>
      
      <CTooltip overlay="Help text" />
      
      <Pagination 
        current={page} 
        total={100} 
        pageSize={10}
        onChange={setPage}
      />
      
      <Dropdown
        value={value}
        options={[
          { label: '选项1', value: '1' },
          { label: '选项2', value: '2' }
        ]}
        onChange={setValue}
      />
      
      <CModal 
        visible={showModal}
        onClose={() => setShowModal(false)}
        title="示例弹窗"
      >
        <div>弹窗内容</div>
      </CModal>
    </div>
  )
}
```
